#include <stdio.h>
#include <string.h>
#include <ctype.h> 

const char* pangrams(const char* s) {
    int seen[26] = {0};  
    int count = 0;       

    for (int i = 0; s[i] != '\0'; i++) {
        char ch = tolower(s[i]);
        if (ch >= 'a' && ch <= 'z') {
            int index = ch - 'a';
            if (!seen[index]) {
                seen[index] = 1;
                count++;
            }
        }
    }

    return (count == 26) ? "pangram" : "not pangram";
}

int main() {
    char s[1001];  
    fgets(s, sizeof(s), stdin);  


    s[strcspn(s, "\n")] = '\0';

    printf("%s\n", pangrams(s));
    return 0;
}
