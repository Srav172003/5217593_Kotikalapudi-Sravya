#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int compare_Asc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}
int compare_Desc(const void *a, const void *b) {
    return (*(int*)b - *(int*)a);
}

const char* twoArrays(int k, int A[], int B[], int n) {
    qsort(A, n, sizeof(int), compare_Asc);
    qsort(B, n, sizeof(int), compare_Desc);

    for (int i = 0; i < n; i++) {
        if (A[i] + B[i] < k) {
            return "NO";
        }
    }
    return "YES";
}

int main() {
    int q;
    scanf("%d", &q);  

    while (q--) {
        int n, k;
        scanf("%d %d", &n, &k);

        int A[n], B[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &A[i]);
        }
        for (int i = 0; i < n; i++) {
            scanf("%d", &B[i]);
        }

        printf("%s\n", twoArrays(k, A, B, n));
    }

    return 0;
}
