#include <stdio.h>
#include <stdlib.h>  

int dia_Difference(int n, int arr[n][n]) {
    int primaryDiagonalSum = 0;
    int secondaryDiagonalSum = 0;

    for (int i = 0; i < n; i++) {
        primaryDiagonalSum += arr[i][i];            
        secondaryDiagonalSum += arr[i][n - i - 1];    
    }

    return abs(primaryDiagonalSum - secondaryDiagonalSum);
}

int main() {
    int n;
    scanf("%d", &n);  

    int arr[n][n];

    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    
    int result = dia_Difference(n, arr);
    printf("%d\n", result);

    return 0;
}
