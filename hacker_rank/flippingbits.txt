#include <stdio.h>
#include <stdint.h>


unsigned int flipping_Bits(unsigned int n) {
    return ~n; 
}

int main() {
    int q;
    scanf("%d", &q); 

    for (int i = 0; i < q; i++) {
        unsigned int n;
        scanf("%u", &n); 
        unsigned int result = flipping_Bits(n);
        printf("%u\n", result); 
    }

    return 0;
}
