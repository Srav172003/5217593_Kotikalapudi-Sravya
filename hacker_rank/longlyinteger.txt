#include <stdio.h>
int lonelyint(int a[], int n) {
    int res = 0;
    for (int i = 0; i < n; i++) {
        res ^= a[i];
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);

    int a[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int res= lonelyint(a, n);
    printf("%d\n", res);

    return 0;
}
