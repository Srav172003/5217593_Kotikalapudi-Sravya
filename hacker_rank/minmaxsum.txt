#include <stdio.h>
#include <limits.h>

void min_Max_Sum(int arr[], int n) {
    long long tot_sum = 0;
    int min = INT_MAX;
    int max = INT_MIN;

    for (int i = 0; i < n; i++) {
        tot_sum += arr[i];
        if (arr[i] < min) min = arr[i];
        if (arr[i] > max) max = arr[i];
    }

    printf("%lld %lld\n", tot_sum - max, tot_sum - min);
}

int main() {
    int arr[5];
    
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    
    min_Max_Sum(arr, 5);
    
    return 0;
}
