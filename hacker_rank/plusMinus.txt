#include <stdio.h>

void plus_minus(int arr[], int n) {
    int pos_cont = 0, neg_cont = 0, zero_cont = 0;
    
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            pos_cont++;
        else if (arr[i] < 0)
            neg_cont++;
        else
            zero_cont++;
    }
    

    printf("%.6f\n", (double)pos_cont / n);
    printf("%.6f\n", (double)neg_cont / n);
    printf("%.6f\n", (double)zero_cont / n);
}

int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    plus_minus(arr, n);
    
    return 0;
}
